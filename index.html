// Ensure raycaster works for gaze events this.el.setAttribute('class', 'clickable'); }, tick: function (time, timeDelta) { const camera = document.querySelector('[camera]'); const camPos = camera.object3D.position; // Floating this.el.object3D.position.y = this.originalPos.y + Math.sin(time / 1000) * this.data.speed; // Parallax effect (small shift based on camera) this.el.object3D.position.x = this.originalPos.x + camPos.x * this.data.parallaxFactor; this.el.object3D.position.z = this.originalPos.z + camPos.z * this.data.parallaxFactor; // Scale animation for gaze const targetScale = this.isGazed ? this.originalScale.clone().multiplyScalar(this.data.scale) : this.originalScale; this.el.object3D.scale.lerp(targetScale, 0.05); }, events: { mouseenter: function () { this.isGazed = true; }.bind(this), mouseleave: function () { this.isGazed = false; }.bind(this) } });
